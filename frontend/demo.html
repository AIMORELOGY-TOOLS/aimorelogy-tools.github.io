📦 快速集成

1. 引入组件

<!-- 在 HTML 中引入 --> <script> const API_BASE_URL = 'https://aimorelogybackend.site'; </script> <script src="wechat-login-component.js"></script>

2. 添加登录按钮

<!-- 登录按钮 --> <button onclick="WechatLogin.open()">微信登录</button> <!-- 或使用内置样式的按钮 --> <button class="wechat-login-btn" onclick="WechatLogin.open()"> <span>微</span> 微信登录 </button>

3. 监听登录事件

// 监听登录成功事件 window.addEventListener('wechatLoginSuccess', function(event) { const { token, openid, loginTime } = event.detail; console.log('登录成功:', { token, openid, loginTime }); // 处理登录成功逻辑 updateUserInterface(); }); // 监听退出登录事件 window.addEventListener('wechatLogout', function() { console.log('用户已退出登录'); // 处理退出登录逻辑 clearUserInterface(); });

4. 检查登录状态

// 检查当前登录状态 const status = WechatLogin.checkStatus(); if (status.isLoggedIn) { console.log('用户已登录:', status.openid); console.log('登录Token:', status.token); } else { console.log('用户未登录'); } // 手动退出登录 WechatLogin.logout();

📚 API 文档

POST /create_qr

创建微信登录二维码

// 响应示例 { "sessionId": "uuid-string", "qrUrl": "https://mp.weixin.qq.com/cgi-bin/showqrcode?ticket=...", "ticket": "ticket-string", "expireSeconds": 600 }
GET /poll?id={sessionId}

轮询登录状态

// 响应示例 { "status": "scanned", // pending | scanned | success "openid": "user-openid", "timestamp": 1234567890 }
POST /finalize_login

完成登录流程

// 请求体 { "sessionId": "uuid-string" } // 响应示例 { "success": true, "token": "login-token", "openid": "user-openid", "loginTime": "2024-01-01T00:00:00.000Z" }
